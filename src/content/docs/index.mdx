## Concurrent Greeting
```hylo
fun quicksort<T is Comparable>(xs: inout T[], start: Int, length: Int) {
  if length < 2 { return }
  let pivot = start + length - 1
  let f = fun[let y = xs[pivot].copy()](x: T) { x < y }
  let midpoint = &xs[start ..< pivot].partition(in_first_half: f)
  &xs.swap_at(midpoint, pivot)
  quicksort(&xs, start, mipoint - start)
  quicksort(&xs, midpoint + 1, length - (midpoint - start) - 1)
}
```
